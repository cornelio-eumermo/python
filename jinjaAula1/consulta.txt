{% for aluno in alunos %}
    <p>{{ aluno.nome }}</p>
{% endfor %}


{% for aluno in alunos %}
    <p>{{ loop.index }} - {{ aluno.nome }}</p>
{% endfor %}

loop.index → começa em 1
loop.index0 → começa em 0
loop.first → primeiro item
loop.last → último item

<html>
<head>
    <title>{% block title %}{% endblock %}</title>
</head>
<body>
    {% block content %}{% endblock %}
</body>
</html>

O filho

{% endblock %}

--------------------------
Filtros básicos:
{{ nome | lower }} → minúsculo
{{ nome | upper }} → maiúsculo
{{ lista | length }} → quantidade de itens
{{ lista | join(", ") }} → unir itens com vírgula e espaço
{{ nome | replace("a", "@") }} → substituir caracteres
{{ lista | sort }} → ordenar itens

--------------------------
